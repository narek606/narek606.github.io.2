        // Создание частиц
        function createParticles() {
            const particles = document.getElementById('particles');
            const particleCount = 50;
            
            for (let i = 0; i < particleCount; i++) {
                const particle = document.createElement('div');
                particle.className = 'particle';
                particle.style.left = Math.random() * 100 + 'vw';
                particle.style.animationDelay = Math.random() * 6 + 's';
                particle.style.animationDuration = (Math.random() * 3 + 3) + 's';
                
                if (Math.random() > 0.5) {
                    particle.style.background = 'var(--neon-yellow)';
                    particle.style.boxShadow = '0 0 6px var(--neon-yellow)';
                } else if (Math.random() > 0.3) {
                    particle.style.background = 'var(--neon-pink)';
                    particle.style.boxShadow = '0 0 6px var(--neon-pink)';
                }
                
                particles.appendChild(particle);
                
                setTimeout(() => {
                    particle.remove();
                    createNewParticle();
                }, parseFloat(particle.style.animationDuration) * 1000);
            }
        }
        
        function createNewParticle() {
            const particles = document.getElementById('particles');
            const particle = document.createElement('div');
            particle.className = 'particle';
            particle.style.left = Math.random() * 100 + 'vw';
            particle.style.animationDuration = (Math.random() * 3 + 3) + 's';
            
            if (Math.random() > 0.5) {
                particle.style.background = 'var(--neon-yellow)';
                particle.style.boxShadow = '0 0 6px var(--neon-yellow)';
            } else if (Math.random() > 0.3) {
                particle.style.background = 'var(--neon-pink)';
                particle.style.boxShadow = '0 0 6px var(--neon-pink)';
            }
            
            particles.appendChild(particle);
            
            setTimeout(() => {
                if (particle.parentNode) {
                    particle.remove();
                    createNewParticle();
                }
            }, parseFloat(particle.style.animationDuration) * 1000);
        }

        // Переключение страниц
        function showPage(pageId) {
            const pages = document.querySelectorAll('.page');
            pages.forEach(page => page.classList.remove('active'));
            document.getElementById(pageId).classList.add('active');
            
            // Анимация появления элементов
            setTimeout(() => {
                const fadeElements = document.querySelectorAll('.fade-in');
                fadeElements.forEach((element, index) => {
                    setTimeout(() => {
                        element.classList.add('visible');
                    }, index * 100);
                });
            }, 100);
        }

        // Калькулятор стоимости
        function updateCalculator() {
            const checkboxes = document.querySelectorAll('.calc-checkbox');
            const totalElement = document.getElementById('total');
            let total = 0;
            
            checkboxes.forEach(checkbox => {
                if (checkbox.checked) {
                    total += parseInt(checkbox.dataset.price);
                }
            });
            
            totalElement.textContent = `Итого: ${total.toLocaleString()}₽`;
            
            if (total > 0) {
                totalElement.style.color = 'var(--neon-yellow)';
                totalElement.style.textShadow = '0 0 10px var(--neon-yellow)';
            } else {
                totalElement.style.color = 'white';
                totalElement.style.textShadow = 'none';
            }
        }

        // Отправка формы
        function sendMessage(event) {
            event.preventDefault();
            
            // Анимация кнопки
            const button = event.target.querySelector('button');
            const originalText = button.textContent;
            button.textContent = 'Отправляется...';
            button.style.background = 'linear-gradient(45deg, var(--cyber-blue), var(--neon-pink))';
            
            setTimeout(() => {
                button.textContent = 'Отправлено! ✅';
                button.style.background = 'linear-gradient(45deg, #00ff00, #00aa00)';
                
                setTimeout(() => {
                    button.textContent = originalText;
                    button.style.background = 'linear-gradient(45deg, var(--neon-yellow), var(--cyber-blue))';
                    event.target.reset();
                }, 2000);
            }, 1500);
        }

        // Анимация при скролле
        function handleScroll() {
            const elements = document.querySelectorAll('.fade-in');
            elements.forEach(element => {
                const elementTop = element.getBoundingClientRect().top;
                const elementVisible = 150;
                
                if (elementTop < window.innerHeight - elementVisible) {
                    element.classList.add('visible');
                }
            });
        }

        // Динамическое изменение фона
        function animateBackground() {
            const body = document.body;
            const time = Date.now() * 0.0001;
            
            const gradient1X = 20 + Math.sin(time) * 10;
            const gradient1Y = 50 + Math.cos(time * 0.7) * 15;
            
            const gradient2X = 80 + Math.sin(time * 0.8) * 12;
            const gradient2Y = 20 + Math.cos(time * 1.2) * 10;
            
            const gradient3X = 40 + Math.sin(time * 1.5) * 8;
            const gradient3Y = 80 + Math.cos(time * 0.9) * 12;
            
            body.style.background = `
                radial-gradient(circle at ${gradient1X}% ${gradient1Y}%, rgba(255, 0, 170, 0.1) 0%, transparent 50%),
                radial-gradient(circle at ${gradient2X}% ${gradient2Y}%, rgba(0, 245, 255, 0.1) 0%, transparent 50%),
                radial-gradient(circle at ${gradient3X}% ${gradient3Y}%, rgba(255, 215, 0, 0.05) 0%, transparent 50%),
                var(--dark-bg)
            `;
            
            requestAnimationFrame(animateBackground);
        }
